        \begin{block}{Erklärkung}
            \begin{itemize}
                \item 
                \begin{equation*}
                    D = t_d \frac{d (W-X)}{dt} = \frac{-X}{dt}
                \end{equation*}
                \item für $T>0$ ist $D < 0$
                \item D-Regler "weiß nichts" vom Sollwert
            \end{itemize}
        \end{block}
