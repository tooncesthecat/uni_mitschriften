    \begin{block}{Theorie}
         \begin{itemize}
            \item linearer Verlauf $\rightarrow$ für Theoriekurve ist
            nur $I_{max}$ nötig
            \begin{equation*}
                I_{max} = \frac{U_{in}}{2R}\left( \frac{1}{8} +
                \frac{1}{4} + \frac{1}{2} + \frac{1}{1} \right) 
                = \frac{15U_{in}}{16R}
                = 458.75 \mu A
            \end{equation*}
            \begin{equation*}
                I_{Theorie} = 458.75 \mu A \cdot x
            \end{equation*}
         \end{itemize}
    \end{block}
\end{frame}
