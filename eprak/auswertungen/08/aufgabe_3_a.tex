\begin{frame}
    \frametitle{Theorie}
    \framesubtitle{}
    \begin{block}{Theoretischer Wert}
        \begin{itemize}
            \item Aufteilung der Spannung $V_{cc}$ in 255 Teile
        \end{itemize}
            \begin{gather*}
                U_{out} = \frac{n}{255} \cdot V_{cc} \\
                n = \left\lfloor \frac{U_{out}}{V_{cc}} \cdot 255 \right\rfloor
            \end{gather*}
    \end{block}
\end{frame}
